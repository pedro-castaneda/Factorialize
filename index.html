		//Select DOM elements into new variables
		let input = $("#input");
		let btn = $("#btn");
		let result= $("#result");


		//Calls Factorial() and passes parsed input value to new variable to be displayed
		btn.on('click',function(){
		let newValue = parseInt(input.val());
		result.text("The Factorial of " + input.val() + " is " + factorial(newValue));
		});


		//This Function takes in a parameter and generates its factorial value.
		function factorial(num){
			let number = num;
			let i = 0;
			let result = 0;

			for (i = num - 1; i >= 1; i--){
		    	num = num * i;
			}
		result = num;
		return result;
		}
